```bash
# Pre-stage (idempotent)
SRC=/Users/andreassjodin/Desktop/snabbsekvensering/bioshield/_data/rawdata-barcodes12
STAGE=/tmp/nanometa_test_src_multiplex
mkdir -p "$STAGE"
for bc in barcode01 barcode02 barcode03 barcode04 barcode05; do
  ln -sfn "$SRC/$bc" "$STAGE/$bc"
done

# Per-scenario outdir
SCEN_OUT=/Users/andreassjodin/Desktop/snabbsekvensering/output-live/audit_realtime_2026_05_09/scenario_1_multiplex
WATCH_DIR="$SCEN_OUT/_watch"
mkdir -p "$WATCH_DIR" "$SCEN_OUT/results" "$SCEN_OUT/work"

# Config
cat > "$SCEN_OUT/config.yaml" <<EOF
nanopore_output_directory: "$WATCH_DIR"
results_output_directory:  "$SCEN_OUT/results"
kraken_db: "/Users/andreassjodin/Desktop/snabbsekvensering/bioshield/kraken_db/k2_pluspfp_08_GB_20251015"
processing_mode: "realtime"
sample_handling: "by_barcode"
pipeline_profile: "conda"
pipeline_source:  "/Users/andreassjodin/Code/nanometanf"
blast_validation: false
update_interval_seconds: 10
realtime_timeout_minutes: 15
EOF

# Dashboard (port 8060 since 8050 may be occupied)
source /Users/andreassjodin/miniforge3/etc/profile.d/conda.sh
conda activate nf-core
python -m nanometa_live.app --config "$SCEN_OUT/config.yaml" --port 8060 \
  > "$SCEN_OUT/dashboard.log" 2>&1 &

# Streaming source
conda activate nanorunner
nanorunner replay \
  --source /tmp/nanometa_test_src_multiplex \
  --target  "$WATCH_DIR" \
  --reads-per-file 250 --interval 15 --batch-size 1 \
  --force-structure multiplex --operation copy \
  > "$SCEN_OUT/nanorunner.log" 2>&1 &

# Pipeline
conda activate nf-core
cd "$SCEN_OUT"
NXF_HOME="$SCEN_OUT/.nextflow" nextflow run /Users/andreassjodin/Code/nanometanf \
  -profile conda \
  --realtime_mode true \
  --nanopore_output_dir "$WATCH_DIR" \
  --outdir              "$SCEN_OUT/results" \
  --kraken2_db          /Users/andreassjodin/Desktop/snabbsekvensering/bioshield/kraken_db/k2_pluspfp_08_GB_20251015 \
  --batch_size 2 --report_write_interval 1 --max_files 30 \
  --realtime_timeout_minutes 10 --enable_blast_validation false \
  -w "$SCEN_OUT/work" -resume 2>&1 | tee "$SCEN_OUT/nextflow.log"

# Open dashboard at http://localhost:8060/
```
